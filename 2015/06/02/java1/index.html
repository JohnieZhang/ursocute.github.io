Java实现一个ATM柜台机系统

Posted on 2015-06-02 | In 代码 |

这学期选了Java的公选课，结课作业是要求我们用Java做图形用户界面的ATM柜员机系统，实现能登陆、取款、存款、查询、修改密码的功能，虽然比较简单，但想到是自己的第一个Java程序，所以还是在博客里稍微记录一下。

import java.awt.*;
import java.awt.event.*;
import javax.swing.*;
class Window3 extends JFrame implements ActionListener{
    JPanel up = new JPanel(new GridLayout(1, 2));
    JPanel middle = new JPanel(new GridLayout(1, 2));
    JPanel down = new JPanel(new GridLayout(1, 2));
    JButton Oldcode = new JButton("Oldcode");
    JButton Newcode = new JButton("Newcode");
    JButton Rewrite = new JButton("Rewrite");    
    JTextField oldcode = new JTextField("");
    JTextField newcode = new JTextField("");
    JTextField rewrite = new JTextField("");    
    Window3(){
        JButton OK = new JButton("OK");
        up.add(Oldcode);
        up.add(oldcode);    
        middle.add(Newcode);
        middle.add(newcode);
        down.add(Rewrite);
        down.add(rewrite);
        getContentPane().setLayout(new GridLayout(4, 1));
        getContentPane().add(up);
        getContentPane().add(middle);
        getContentPane().add(down);    
        getContentPane().add(OK);            
          OK.addActionListener(this);    
        setVisible(true);
        pack();
    }    
    public void actionPerformed(ActionEvent e) {
        String a1 = oldcode.getText();
        String b1 = newcode.getText();
        String c1 = rewrite.getText();
        if (!a1.equals(Window1.b)) {
            JOptionPane.showMessageDialog(null,"wrong code");
            }
        if (!b1.equals(c1)) {
            JOptionPane.showMessageDialog(null,"the code you rewrite is wrong ");
            }
        if (b1.equals("111111")||b1.equals("222222")||b1.equals("333333")||b1.equals("444444")||b1.equals("555555")||b1.equals("666666")||b1.equals("777777")||b1.equals("888888")||b1.equals("999999")||b1.equals("000000")) {
            JOptionPane.showMessageDialog(null,"it si not a safe code");
            }    
        if (a1.equals(Window1.b)){    
            while (!a1.equals(Window1.b));        
            Window1.b=b1;
            setVisible(false);
            }
    }
}

class Window2 extends JFrame implements ActionListener{
    JButton Inquiry = new JButton("Inquiry");
    JButton Deposit = new JButton("Deposit");
    JButton Draw = new JButton("Draw");
    JButton Revise = new JButton("Revise");
    Window2(){
        Inquiry.addActionListener(this);
        Deposit.addActionListener(this);        
        Draw.addActionListener(this);
        Revise.addActionListener(this);        
        getContentPane().setLayout(new GridLayout(4, 1));
        getContentPane().add(Inquiry);
        getContentPane().add(Deposit);
        getContentPane().add(Draw);    
        getContentPane().add(Revise);            
        pack();
        setVisible(true);
    }
    public void actionPerformed(ActionEvent e) {
        if (e.getSource().equals(Inquiry)) {
            JOptionPane.showMessageDialog(null, "you have $"+Window1.money);  
        }    
        if (e.getSource().equals(Deposit)) {
            String str=JOptionPane.showInputDialog(null,"input the money\n");  
            int i=Integer.parseInt(str);
            while (i%100!=0||i<=0) {
                JOptionPane.showMessageDialog(null,"you need input the right number");
                str=JOptionPane.showInputDialog(null,"input the money\n");  
                i=Integer.parseInt(str);
            }
            Window1.money+=i;
        }
        if (e.getSource().equals(Draw)) {
            String str1=JOptionPane.showInputDialog(null,"input the money\n");  
            int j=Integer.parseInt(str1);
            while (j%100!=0||j<=0) {
                JOptionPane.showMessageDialog(null,"you need input the right number");
                str1=JOptionPane.showInputDialog(null,"input the money\n");  
                j=Integer.parseInt(str1);
            }
            Window1.money-=j;  
        }
        if (e.getSource().equals(Revise)) {
            new Window3(); 
        }            
    }
}

}
